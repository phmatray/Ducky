# Layer 2: Source Generators Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Architecture:** US-08 (IAction) must be done first as US-09 and US-10 depend on it. US-09 and US-10 are independent and can be parallelized.

**Tech Stack:** .NET 9, C# 13, Roslyn Incremental Generators (netstandard2.0), xUnit

**Issues:** #219, #220, #221

- [ ] **Step 4: Create ReducerSourceGenerator**

Create `src/library/Ducky.Generator/ReducerSourceGenerator.cs`:

```csharp
using System.Collections.Generic;
using System.Collections.Immutable;
using System.Linq;
using System.Text;
using Ducky.Generator.Sources;
using Microsoft.CodeAnalysis;
using Microsoft.CodeAnalysis.CSharp;
using Microsoft.CodeAnalysis.CSharp.Syntax;

namespace Ducky.Generator;

[Generator]
public class ReducerSourceGenerator : SourceGeneratorBase
{
    private const string GeneratorNamespace = "Ducky.Generators";
    private const string AttributeName = "DuckyReducerAttribute";
    private const string InitialStateAttrName = "InitialStateAttribute";

    public override void Initialize(IncrementalGeneratorInitializationContext context)
    {
        context.RegisterPostInitializationOutput(ctx =>
        {
            var attrSource = new DuckyReducerAttributeSource(GeneratorNamespace, AttributeName);
            ctx.AddSource($"{AttributeName}.g.cs", attrSource.ToSourceText());

            var initSource = new InitialStateAttributeSource(GeneratorNamespace, InitialStateAttrName);
            ctx.AddSource($"{InitialStateAttrName}.g.cs", initSource.ToSourceText());
        });

        IncrementalValuesProvider<ClassDeclarationSyntax> classProvider = context.SyntaxProvider
            .CreateSyntaxProvider(
                static (node, _) => node is ClassDeclarationSyntax { AttributeLists.Count: > 0 } cls
                    && cls.Modifiers.Any(SyntaxKind.StaticKeyword)
                    && cls.Modifiers.Any(SyntaxKind.PartialKeyword),
                static (ctx, _) => GetReducerClass(ctx))
            .Where(result => result is not null)
            .Select((result, _) => result!);

        context.RegisterSourceOutput(
            context.CompilationProvider.Combine(classProvider.Collect()),
            (spc, source) => GenerateReducers(source.Left, source.Right, spc));
    }

    private static ClassDeclarationSyntax? GetReducerClass(GeneratorSyntaxContext context)
    {
        var classDecl = (ClassDeclarationSyntax)context.Node;

        foreach (AttributeListSyntax attrList in classDecl.AttributeLists)
        {
            foreach (AttributeSyntax attr in attrList.Attributes)
            {
                SymbolInfo symbolInfo = context.SemanticModel.GetSymbolInfo(attr);
                ISymbol? targetSymbol = symbolInfo.Symbol ?? symbolInfo.CandidateSymbols.FirstOrDefault();
                if (targetSymbol is IMethodSymbol attrCtor
                    && attrCtor.ContainingType.ToDisplayString() == $"{GeneratorNamespace}.{AttributeName}")
                {
                    return classDecl;
                }
            }
        }

        return null;
    }

    private void GenerateReducers(
        Compilation compilation,
        ImmutableArray<ClassDeclarationSyntax> classes,
        SourceProductionContext context)
    {
        foreach (ClassDeclarationSyntax classSyntax in classes)
        {
            SemanticModel model = compilation.GetSemanticModel(classSyntax.SyntaxTree);
            if (model.GetDeclaredSymbol(classSyntax) is not INamedTypeSymbol classSymbol)
                continue;

            // Find all static On() methods
            var onMethods = classSymbol.GetMembers()
                .OfType<IMethodSymbol>()
                .Where(m => m.IsStatic && m.Name == "On" && m.Parameters.Length == 2)
                .ToList();

            if (onMethods.Count == 0)
            {
                context.ReportDiagnostic(Diagnostic.Create(
                    new DiagnosticDescriptor("DUCKY001", "No On() methods",
                        "DuckyReducer class '{0}' has no static On(TState, TAction) methods.",
                        "Ducky", DiagnosticSeverity.Warning, true),
                    classSyntax.GetLocation(), classSymbol.Name));
                continue;
            }

            // Infer state type from first parameter of first On() method
            ITypeSymbol stateType = onMethods[0].Parameters[0].Type;
            string stateTypeFqn = stateType.ToDisplayString(SymbolDisplayFormat.FullyQualifiedFormat);
            string className = classSymbol.Name;
            string classNamespace = classSymbol.ContainingNamespace.ToDisplayString();
            string sliceClassName = $"{className}Slice";

            // Validate all On() methods use the same state type
            foreach (IMethodSymbol method in onMethods)
            {
                if (!SymbolEqualityComparer.Default.Equals(method.Parameters[0].Type, stateType))
                {
                    context.ReportDiagnostic(Diagnostic.Create(
                        new DiagnosticDescriptor("DUCKY002", "Inconsistent state type",
                            "All On() methods in '{0}' must have the same first parameter type. Expected '{1}', got '{2}'.",
                            "Ducky", DiagnosticSeverity.Error, true),
                        method.Locations[0], className,
                        stateType.Name, method.Parameters[0].Type.Name));
                    return;
                }
            }

            // Find [InitialState] member
            string initialStateExpr = $"new {stateTypeFqn}()";
            foreach (ISymbol member in classSymbol.GetMembers())
            {
                if (member is IFieldSymbol or IPropertySymbol)
                {
                    bool hasInitialState = member.GetAttributes()
                        .Any(a => a.AttributeClass?.Name == InitialStateAttrName);
                    if (hasInitialState)
                    {
                        initialStateExpr = $"{classSymbol.ToDisplayString(SymbolDisplayFormat.FullyQualifiedFormat)}.{member.Name}";
                        break;
                    }
                }
            }

            // Generate On() registrations
            StringBuilder registrations = new();
            foreach (IMethodSymbol method in onMethods)
            {
                string actionTypeFqn = method.Parameters[1].Type
                    .ToDisplayString(SymbolDisplayFormat.FullyQualifiedFormat);
                string classFqn = classSymbol.ToDisplayString(SymbolDisplayFormat.FullyQualifiedFormat);
                registrations.AppendLine(
                    $"        On<{actionTypeFqn}>((state, action) => {classFqn}.On(state, action));");
            }

            // Generate the SliceReducers subclass
            string code = $@"// <auto-generated />
// Generated by Ducky [DuckyReducer] source generator.

namespace {classNamespace};

/// <summary>
/// Auto-generated SliceReducers for {className}.
/// </summary>
public sealed class {sliceClassName} : Ducky.SliceReducers<{stateTypeFqn}>
{{
    public {sliceClassName}()
    {{
{registrations}    }}

    public override {stateTypeFqn} GetInitialState()
    {{
        return {initialStateExpr};
    }}
}}
";

            AddSource(context, $"{sliceClassName}.g.cs", code);
        }
    }
}
```

- [ ] **Step 3: Create EffectSourceGenerator**

Create `src/library/Ducky.Generator/EffectSourceGenerator.cs`:

```csharp
using System.Collections.Generic;
using System.Collections.Immutable;
using System.Linq;
using System.Text;
using Ducky.Generator.Sources;
using Microsoft.CodeAnalysis;
using Microsoft.CodeAnalysis.CSharp;
using Microsoft.CodeAnalysis.CSharp.Syntax;

namespace Ducky.Generator;

[Generator]
public class EffectSourceGenerator : SourceGeneratorBase
{
    private const string GeneratorNamespace = "Ducky.Generators";
    private const string AttributeName = "DuckyEffectAttribute";

    private static readonly HashSet<string> FrameworkTypes = new()
    {
        "Ducky.IStateProvider",
        "Ducky.IDispatcher",
        "System.Threading.CancellationToken"
    };

    public override void Initialize(IncrementalGeneratorInitializationContext context)
    {
        context.RegisterPostInitializationOutput(ctx =>
        {
            var attrSource = new DuckyEffectAttributeSource(GeneratorNamespace, AttributeName);
            ctx.AddSource($"{AttributeName}.g.cs", attrSource.ToSourceText());
        });

        IncrementalValuesProvider<ClassDeclarationSyntax> classProvider = context.SyntaxProvider
            .CreateSyntaxProvider(
                static (node, _) => node is ClassDeclarationSyntax { AttributeLists.Count: > 0 } cls
                    && cls.Modifiers.Any(SyntaxKind.StaticKeyword)
                    && cls.Modifiers.Any(SyntaxKind.PartialKeyword),
                static (ctx, _) => GetEffectClass(ctx))
            .Where(result => result is not null)
            .Select((result, _) => result!);

        context.RegisterSourceOutput(
            context.CompilationProvider.Combine(classProvider.Collect()),
            (spc, source) => GenerateEffects(source.Left, source.Right, spc));
    }

    private static ClassDeclarationSyntax? GetEffectClass(GeneratorSyntaxContext context)
    {
        var classDecl = (ClassDeclarationSyntax)context.Node;

        foreach (AttributeListSyntax attrList in classDecl.AttributeLists)
        {
            foreach (AttributeSyntax attr in attrList.Attributes)
            {
                SymbolInfo symbolInfo = context.SemanticModel.GetSymbolInfo(attr);
                ISymbol? targetSymbol = symbolInfo.Symbol ?? symbolInfo.CandidateSymbols.FirstOrDefault();
                if (targetSymbol is IMethodSymbol attrCtor
                    && attrCtor.ContainingType.ToDisplayString() == $"{GeneratorNamespace}.{AttributeName}")
                {
                    return classDecl;
                }
            }
        }

        return null;
    }

    private void GenerateEffects(
        Compilation compilation,
        ImmutableArray<ClassDeclarationSyntax> classes,
        SourceProductionContext context)
    {
        foreach (ClassDeclarationSyntax classSyntax in classes)
        {
            SemanticModel model = compilation.GetSemanticModel(classSyntax.SyntaxTree);
            if (model.GetDeclaredSymbol(classSyntax) is not INamedTypeSymbol classSymbol)
                continue;

            string classNamespace = classSymbol.ContainingNamespace.ToDisplayString();
            string classFqn = classSymbol.ToDisplayString(SymbolDisplayFormat.FullyQualifiedFormat);

            // Find all static async methods returning Task
            var methods = classSymbol.GetMembers()
                .OfType<IMethodSymbol>()
                .Where(m => m.IsStatic
                    && m.ReturnsVoid == false
                    && m.ReturnType.ToDisplayString() == "System.Threading.Tasks.Task"
                    && m.Parameters.Length >= 1)
                .ToList();

            foreach (IMethodSymbol method in methods)
            {
                IParameterSymbol actionParam = method.Parameters[0];
                string actionTypeFqn = actionParam.Type.ToDisplayString(SymbolDisplayFormat.FullyQualifiedFormat);
                string effectClassName = $"{classSymbol.Name}_{method.Name}_Effect";

                // Classify parameters
                List<IParameterSymbol> diParams = new();
                foreach (IParameterSymbol param in method.Parameters.Skip(1))
                {
                    string paramTypeFqn = param.Type.ToDisplayString();
                    if (!FrameworkTypes.Contains(paramTypeFqn))
                    {
                        diParams.Add(param);
                    }
                }

                // Build DI fields and constructor params
                StringBuilder diFields = new();
                StringBuilder ctorParams = new();
                StringBuilder ctorAssignments = new();
                foreach (IParameterSymbol diParam in diParams)
                {
                    string typeFqn = diParam.Type.ToDisplayString(SymbolDisplayFormat.FullyQualifiedFormat);
                    string fieldName = $"_{diParam.Name}";
                    diFields.AppendLine($"    private readonly {typeFqn} {fieldName};");
                    ctorParams.Append($", {typeFqn} {diParam.Name}");
                    ctorAssignments.AppendLine($"        {fieldName} = {diParam.Name};");
                }

                // Build method call arguments
                StringBuilder callArgs = new();
                callArgs.Append("action");
                foreach (IParameterSymbol param in method.Parameters.Skip(1))
                {
                    string paramTypeFqn = param.Type.ToDisplayString();
                    callArgs.Append(", ");
                    if (paramTypeFqn == "Ducky.IStateProvider")
                        callArgs.Append("stateProvider");
                    else if (paramTypeFqn == "Ducky.IDispatcher")
                        callArgs.Append("Dispatcher");
                    else if (paramTypeFqn == "System.Threading.CancellationToken")
                        callArgs.Append("token");
                    else
                        callArgs.Append($"_{param.Name}");
                }

                string code = $@"// <auto-generated />
// Generated by Ducky [DuckyEffect] source generator.

namespace {classNamespace};

/// <summary>
/// Auto-generated AsyncEffect for {classSymbol.Name}.{method.Name}.
/// </summary>
public sealed class {effectClassName} : Ducky.AsyncEffect<{actionTypeFqn}>
{{
{diFields}
    public {effectClassName}({(diParams.Count > 0 ? ctorParams.ToString().TrimStart(',', ' ') : "")})
    {{
{ctorAssignments}    }}

    public override async System.Threading.Tasks.Task HandleAsync(
        {actionTypeFqn} action,
        Ducky.IStateProvider stateProvider,
        System.Threading.CancellationToken token)
    {{
        await {classFqn}.{method.Name}({callArgs});
    }}
}}
";

                AddSource(context, $"{effectClassName}.g.cs", code);
            }
        }
    }
}
```
